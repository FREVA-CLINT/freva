14class SolrFindFiles(object): 

15 """Encapsulate access to Solr like the find files command""" 

16 def __init__(self, core=None, host=None, port=None, get_status=False): 

17 """Create the connection pointing to the proper solr url and core.  

18The default values of these parameters are setup in evaluation_system.model.solr_core.SolrCore

20 

21:param core: name of the solr core that will be used. 

22:param host: hostname of the machine where the solr core is to be found. 

23:param port: port number of the machine where the solr core is to be found. 

24:param get_status: if the core should be contacted in an attempt to get more metadata. 

25""" 

26 self.solr = SolrCore(core, host=host, port=port, get_status=get_status)

29 return '<SolrFindFiles %s>' % self.solr

31 def _to_solr_query(self, partial_dict): 

32 """Creates a Solr query assuming the default operator is "AND". See schema.xml for that.""" 

33 params = [] 

34 # these are special Solr keys that we might get and we assume are not meant for the search 

35 special_keys = "q fl fq facet.limit".split() 

36 

37 for key, value in partial_dict.items(): 

38 if key in special_keys: 

39 params.append((key, value)) 

40 else: 

41 if key.endswith('_not_'): 

42 # handle negation 

43 key = '-' + key[:-5] 

44 if isinstance(value, list): 

45 # implies an or 

46 constraint = ' OR '.join(['%s:%s' % (key, v) for v in value]) 

47 else: 

48 constraint = '%s:%s' % (key, value) 

49 params.append(('fq', constraint,)) 

50 return urllib.parse.urlencode(params)

52 def _search(self, batch_size=10000, latest_version=False, _retrieve_metadata=False, **partial_dict): 

53 """This encapsulates the Solr call to get documents and returns an iterator providing the. The special 

54parameter _retrieve_metadata will affect the first value returned by the iterator.  

55 

56:param batch_size: the amount of files to be buffered from Solr. 

57:param latest_version: if the search should *try* to find the latest version from all contained here. Please note 

58 that we don't use this anymore. Instead we have 2 cores and this is defined directly in :class:`SolrFindFiles.search`. 

59 It was changed because it was slow and required too much memory. 

60:param _retrieve_metadata: if set to true, the first item on the iterator is a metadata one. This is used so it can be 

61known beforehand how many values are going to be returned, even before getting them all. To avoid this we might 

62implement a result set object. But that would break the find_files compatibility. 

63""" 

64 offset = partial_dict.pop('start', 0) 

65 # value retrieved from sys.maxint and == 2**31-1 

66 max_rows = partial_dict.pop('rows', 2147483647) 

67 

68 if 'text' in partial_dict: 

69 partial_dict.update({'q': partial_dict.pop('text')}) 

70 else: 

71 partial_dict.update({'q': '*:*'}) 

72 

73 if 'fl' not in partial_dict: 

74 partial_dict.update({'fl': 'file'}) 

75 

76 query = self._to_solr_query(partial_dict) 

77 # DEPRECATED: 

78 # This is not used anymore, because we have 2 different cores now 

79 if latest_version: # pragma nocover 

80 query += '&group=true&group.field=file_no_version&group.sort=version+desc&group.ngroups=true&group.format=simple' 

81 

82 while True: 

83 if max_rows < batch_size: 

84 batch_size = max_rows 

85 answer = self.solr.get_json('select?start=%s&rows=%s&%s' % (offset, batch_size, query)) 

86 if _retrieve_metadata: 

87 meta = answer['response'].copy() 

88 del meta['docs'] 

89 yield meta 

90 _retrieve_metadata = False 

91 # Not used anymore (see above) 

92 if latest_version: # pragma nocover 

93 offset = answer['grouped']['file_no_version']['doclist']['start'] 

94 total = answer['grouped']['file_no_version']['ngroups'] 

95 iter_answer = answer['grouped']['file_no_version']['doclist']['docs'] 

96 else: 

97 offset = answer['response']['start'] 

98 total = answer['response']['numFound'] 

99 iter_answer = answer['response']['docs'] 

100 

101 for item in iter_answer: 

102 yield item['file'] 

103 

104 max_rows -= total 

105 if total - offset <= batch_size or max_rows <= 0: 

106 break # we are done 

107 else: 

108 offset += batch_size

111 def search(latest_version=True, **partial_dict): 

112 """It mimics the same :class:`evaluation_system.model.file.DRSFile.search` behavior. 

113The implementation contacts the required Solr cores instead of contacting the file system. 

114 

115:param latest_version: defines if looking for the latest version of a file only, or for any. 

116:param partial_dict: the search dictionary for solr. It might also contain some special values as 

117 defined in :class:`SolrFindFiles._search` 

118:returns: An iterator over the results. 

119""" 

120 # use defaults, if other required use _search in the SolrFindFiles instance 

121 if latest_version: 

122 s = SolrFindFiles(core='latest') 

123 else: 

124 s = SolrFindFiles(core='files') 

125 return s._search(latest_version=False, **partial_dict)

127 def _facets(self, latest_version=False, facets=None, **partial_dict): 

128 if facets and not isinstance(facets, list): 

129 if ',' in facets: 

130 # we assume multiple values here 

131 facets = [f.strip() for f in facets.split(',')] 

132 else: 

133 facets = [facets] 

134 

135 if 'text' in partial_dict: 

136 partial_dict.update({'q': partial_dict.pop('text')}) 

137 else: 

138 partial_dict.update({'q': '*:*'}) 

139 

140 query = self._to_solr_query(partial_dict) 

141 

142 if facets is None: 

143 # get all minus what we don't want 

144 facets = set(self.solr.get_solr_fields())\ 

145 - set(['', '_version_', 'file_no_version', 'level', 'timestamp', 

146 'time', 'creation_time', 'source', 'version', 'file', 

147 'file_name']) 

148 

149 if facets: 

150 query += '&facet=true&facet.sort=index&facet.mincount=1&facet.field=' + '&facet.field='.join(facets)

152 if latest_version: # pragma nocover (see above) 

153 query += '&group=true&group.field=file_no_version&group.facet=true' 

154 

155 answer = self.solr.get_json('select?facet=true&rows=0&%s' % query)
