35class PluginAbstract(metaclass=abc.ABCMeta): 

36 """This is the base class for all plug-ins. It is the only class that needs to be inherited from when implementing a plug-in. 

37  

38From it, you'll need to implement the few attributes and/or methods marked as abstract with the decorator 

39``@abc.abstractproperty`` or ``@abc.abstractmethod``. If you don't you'll get a message informing you which methods 

40and or variables need to be implemented. Refer to their documentation to know what they should do. 

41 

42As usual, you may overwrite all methods and properties defined in here, but you'll be breaking the contract 

43between the methods so you'll have to make sure it doesn't break anything else. Please write some tests 

44for your own class that checks it is working as expected. The best practice is to use what is provided as is and only 

45implement what is required (and more, if you need, just don't overwrite any methods/variable if you don't need to) 

46 

47This very short example shows a complete plug-in. Although it does nothing it already show the most important part, 

48the :class:`evaluation_system.api.parameters.ParameterDictionary` used for defining meta-data on the parameters:: 

49 

50 from evaluation_system.api import plugin, parameters 

51  

52 class MyPlugin(plugin.PluginAbstract): 

53 __short_description__ = "MyPlugin short description (just to know what it does)"  

54 __version__ = (0,0,1) 

55 __parameters__ = parameters.ParameterDictionary( 

56 parameters.Integer(name='number', help='This is an optional configurable int variable named number without default value and this description'), 

57 parameters.Float(name='the_number',mandatory=True, help='A required float value without default'), 

58 parameters.Bool(name='really', default=False, help='a boolean parameter named really with default value of false'),  

59 parameters.String(name='str')) #a simple optional string without any other information 

60  

61 def runTool(self, config_dict=None): 

62 print "MyPlugin", config_dict 

63 

64If you need to test it use the ``EVALUATION_SYSTEM_PLUGINS`` environmental variable to point to the source directory and package. 

65For example assuming you have th source code in ``/path/to/source`` and the package holding the class implementing :class:`evaluation_system.api.plugin` is 

66``package.plugin_module`` (i.e. its absolute file path is ``/path/to/source/package/plugin_module.py``), you would tell the system how to find 

67the plug-in by issuing the following command (bash & co):: 

68 

69 export EVALUATION_SYSTEM_PLUGINS=/path/to/source,package.plugin_module 

70 

71Use a colon to separate multiple items:: 

72 

73 export EVALUATION_SYSTEM_PLUGINS=/path1,plguin1:/path2,plugin2:/path3,plugin3 

74 

75By telling the system where to find the packages it can find the :class:`evaluation_system.api.plugin` implementations. The system just loads the packages and get to the classes using the :py:meth:`class.__subclasses__` method. The reference speaks about *weak references* so it's not clear if (and when) they get removed.  

76We might have to change this in the future if it's not enough. Another approach would be forcing self-registration of a class in the :ref:`__metaclass__ <python:datamodel>` attribute when the class is implemented.  

77 

78For more general (and less technical) information refer to the wiki: https://code.zmaw.de/projects/miklip-d-integration/wiki 

79"""

246 def prepareOutput(self, output_files): 

247 """Prepare output for files supposedly created. This method checks the files exist 

248and returns a dictionary with information about them:: 

249 

250 { <absolute_path_to_file>: { 

251 'timestamp': os.path.getctime(<absolute_path_to_file>), 

252 'size': os.path.getsize(<absolute_path_to_file>) 

253 } 

254 } 

255 

256Use it for the return call of runTool. 

257  

258:param output_files: iterable of strings or single string with paths to all files that where created by the tool. 

259:return: dictionary with the paths to the files that were created as key and a dictionary as value. 

260 """ 

261 result = {} 

262 if isinstance(output_files, str): output_files = [output_files] 

263 for file_path in output_files: 

264 metadata = {} 

265 

266 # we expect a meta data dictionary 

267 if isinstance(output_files, dict): 

268 metadata = output_files[file_path] 

269 if not isinstance(metadata, dict): 

270 raise ValueError('Meta information must be of type dict') 

271 

272 if os.path.isfile(file_path): 

273 self._extend_output_metadata(file_path, metadata) 

274 result[os.path.abspath(file_path)] = metadata 

275 elif os.path.isdir(file_path): 

276 #ok, we got a directory, so parse the contents recursively 

277 for file_path in [os.path.join(r,f) for r,_,files in os.walk(file_path) for f in files]: 

278 filemetadata = metadata.copy() 

279 self._extend_output_metadata(file_path, filemetadata) 

280 

281 # update meta data with user entries 

282 usermetadata = result.get(os.path.abspath(file_path), {}) 

283 filemetadata.update(usermetadata) 

284 

285 result[os.path.abspath(file_path)] = filemetadata 

286 else: 

287 result[os.path.abspath(file_path)] = metadata 

288 return result

290 def _extend_output_metadata(self, file_path, metadata): 

291 fstat=os.stat(file_path) 

292 

293 if 'timestamp' not in metadata: 

294 metadata['timestamp'] = fstat[stat.ST_CTIME] 

295 # metadata['timestamp'] = os.path.getctime(file_path) 

296 if 'size' not in metadata: 

297 metadata['size'] = fstat[stat.ST_SIZE] 

298 # metadata['size'] = os.path.getsize(file_path) 

299 if 'type' not in metadata: 

300 ext = os.path.splitext(file_path) 

301 if ext: 

302 ext = ext[-1].lower() 

303 if ext in '.jpg .jpeg .png .gif'.split(): 

304 metadata['type'] = 'plot' 

305 metadata['todo'] = 'copy' 

306 

307 if ext in '.tif .svg .ps .eps'.split(): 

308 metadata['type'] = 'plot' 

309 metadata['todo'] = 'convert' 

310 

311 if ext == '.pdf': 

312 #If pdfs have more than one page we don't convert them,  

313 #instead we offer a download link 

314 pdf = PdfFileReader(open(file_path,'rb')) 

315 num_pages = pdf.getNumPages() 

316 metadata['type'] = 'pdf' 

317 if num_pages > 1: 

318 metadata['todo'] = 'copy' 

319 else: 

320 metadata['todo'] = 'convert' 

321 

322 if ext in '.tex'.split(): 

323 metadata['type'] = 'plot' 

324 

325 elif ext in '.nc .bin .ascii'.split(): 

326 metadata['type'] = 'data' 

327 if ext in ['.zip']: 

328 metadata['type'] = 'pdf' 

329 metadata['todo'] = 'copy'

398 def _parseConfigStrValue(self, param_name, str_value, fail_on_missing=True): 

399 """Try to parse the string in ``str_value`` into the most appropriate value according  

400to the parameter logic. 

401 

402The string *None* will be mapped to the value ``None``. On the other hand the quoted word *"None"* will remain as the 

403string ``"None"`` without any quotes. 

404 

405:param param_name: Parameter name to which the string belongs. 

406:type param_name: str 

407:param str_value: The string that will be parsed. 

408:type str_value: str 

409:param fail_on_missing: If the an exception should be risen in case the param_name is not found in :class:`__parameters__` 

410:type fail_on_missing: bool 

411:return: the parsed string, or the string itself if it couldn't be parsed, but no exception was thrown. 

412:raises: ( :class:`ConfigurationError` ) if parsing couldn't succeed. 

413""" 

414 

415 if str_value == "None": return None 

416 elif str_value == '"None"': str_value = "None" 

417 

418 if self.__parameters__ is None or (not fail_on_missing and param_name not in self.__parameters__): 

419 #if there's no dictionary reference or the param_name is not in it and we are not failing 

420 #just return the str_value  

421 return str_value 

422 

423 else: 

424 return self.__parameters__.get_parameter(param_name).parse(str_value)

428 def setupConfiguration(self, config_dict = None, check_cfg = True, recursion=True, substitute=True): 

429 """Defines the configuration required for running this plug-in.  

430If not a dictionary will be returned but with all indirections being resolved. E.g.:: 

431 

432 dict(a=1, b='1.txt', c='old_1.txt') == setupConfiguration(config_dict=dict(a=1, b='$a.txt', c='old_$b')) 

433 

434Basically the default values from :class:`__parameters__` will be updated with the values from ``config_dict``. 

435There are some special values pointing to user-related managed by the system defined in :class:`evaluation_system.model.user.User.getUserVarDict` . 

436  

437:param config_dic: dictionary with the configuration to be used when generating the configuration file. 

438:type config_dic: dict or :class:`metadict` 

439:param check_cfg: whether the method checks that the resulting configuration dictionary (i.e. the default  

440 updated by `config_dict`) has no None values after all substituions are made. 

441:type check_cfg: bool 

442:param recursion: Whether when resolving the template recursion will be applied, i.e. variables can be set 

443 with the values of other variables, e.g. ``recursion && a==1 && b=="x${a}x" => f(b)=="x1x"``  

444:type recursion: bool 

445:return: a copy of self.self.__config_metadict__ with all defaults values plus those provided here. 

446 """ 

447 if config_dict: 

448 conf = dict(self.__parameters__) 

449 conf.update(config_dict) 

450 config_dict = conf 

451 else: 

452 config_dict = dict(self.__parameters__) 

453 

454 if substitute: 

455 results = self._special_variables.substitute(config_dict, recursive=recursion) 

456 else: 

457 results = config_dict.copy() 

458 

459 if check_cfg: 

460 self.__parameters__.validate_errors(results, raise_exception=True) 

461 

462 return results

464 def readFromConfigParser(self, config_parser): 

465 """Reads a configuration from a config parser object. 

466The values are assumed to be in a section named just like the class implementing this method. 

467  

468:param config_parser: From where the configuration is going to be read. 

469:type config_parser: ConfigParser.RawConfigParser 

470:return: a :class:`metadict` which is a clone of :class:`__config_metadict__` (if available) updated with the 

471 information found in the config Parser""" 

472 

473 section = self.__class__.__name__ 

474 #create a copy of metadict 

475 result = dict(self.__parameters__) 

476 #we do this to avoid having problems with the "DEFAULT" section as it might define 

477 #more options that what this plugin requires 

478 keys = set(result).intersection(config_parser.options(section)) 

479 #update values as found in the configuration 

480 for key in keys: 

481 #parse the value as good as possible 

482 result[key] = self._parseConfigStrValue(key, config_parser.get(section, key)) 

483 return result

495 def saveConfiguration(self, fp, config_dict=None, include_defaults=False): 

496 """Stores the given configuration to the provided file object. 

497if no configuration is provided the default one will be used. 

498 

499:param fp: An object with a readline argument (e.g. as return by :py:func:`open` ) from where the configuration is going to be read. 

500:param config_dict: a metadict with the configuration to be stored. If none is provided the result from 

501 :class:`setupConfiguration` with ``check_cfg=False`` will be used.""" 

502 #store the section header 

503 if config_dict is None: 

504 #a default incomplete one 

505 config_dict = self.setupConfiguration(check_cfg = False, substitute=False) 

506 fp.write('[%s]\n' % self.__class__.__name__) 

507 

508 import textwrap 

509 wrapper = textwrap.TextWrapper(width=80, initial_indent='#: ', subsequent_indent='#: ', replace_whitespace=False,break_on_hyphens=False,expand_tabs=False) 

510 

511 #preserve order 

512 for param_name in self.__parameters__: 

513 if include_defaults: 

514 param = self.__parameters__.get_parameter(param_name) 

515 if param.help: 

516 #make sure all new lines are comments! 

517 help_lines = param.help.splitlines() 

518 if param.mandatory: 

519 help_lines[0] = '[mandatory] ' + help_lines[0] 

520 fp.write('\n'.join([wrapper.fill(line) for line in help_lines])) 

521 fp.write('\n') 

522 value = config_dict.get(param_name, None) 

523 if value is None: 

524 #means this is not setup 

525 if param.mandatory: 

526 value="<THIS MUST BE DEFINED!>"

613 def composeCommand(self, config_dict=None, scheduled_id=None, 

614 batchmode=False, email=None, caption=None, 

615 unique_output=True): 

616 logging.debug('config dict:' + str(config_dict)) 

617 logging.debug('scheduled_id:' + str(scheduled_id)) 

618 

619 cmd_param = 'analyze ' 

620 cmd_param = 'freva --plugin ' 

621 cmd_param += self.__class__.__name__ 

622 # write explicitly if batchmode is requested 

623 cmd_param += ' --batchmode=%s' % str(batchmode) 

624 

625 # add a given e-mail 

626 if email: 

627 cmd_param += ' --mail=%s' % email 

628 

629 # the parameter string 

630 #cmd_param += ' --tool ' + self.__class__.__name__ 

631 

632 

633 # add a caption if given 

634 if not caption is None: 

635 quote_caption = caption 

636 quote_caption = caption.replace("\\", "\\\\") 

637 quote_caption = quote_caption.replace("'", "'\\''") 

638 cmd_param += " --caption '%s'" % quote_caption 

639 

640 # append the unique_output param 

641 cmd_param += " --unique_output %s" % unique_output 

642 

643 # a scheduled id overrides the dictionary behavior 

644 if scheduled_id: 

645 cmd_param += ' --scheduled-id %i' % scheduled_id 

646 

647 else: 

648 #store the section header 

649 if config_dict is None: 

650 #a default incomplete one 

651 config_dict = self.setupConfiguration(check_cfg = False, substitute=False) 

652 else: 

653 config_dict = self.setupConfiguration(config_dict = config_dict, check_cfg = False, substitute=False)

655 # compose the parameters preserve order 

656 for param_name in self.__parameters__: 

657 if param_name in config_dict: 

658 param = self.__parameters__.get_parameter(param_name) 

659 value = config_dict[param_name] 

660 isMandatory = param.mandatory 

661 

662 if value is None: 

663 if isMandatory: 

664 raise self.ExceptionMissingParam(param_name) 

665 else: 

666 cmd_param += " %s=%s" % (param_name, param.str(value)) 

667 

668 logging.debug('Execute command:' + cmd_param) 

669 

670 return cmd_param
