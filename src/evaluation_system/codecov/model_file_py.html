142 def to_dataset(self, versioned=False, to_path=False): 

143 """creates the dataset to which this file is part of out of the DRS information. 

144 

145:param versioned: If the dataset should contain information about the version. Note that not 

146 all DRS structures are versioned, so in those cases where there is just no 

147 version information this makes no difference. 

148:type versioned: bool 

149:param to_path: if true return the path to the dataset, otherwise returns the dataset identifier.""" 

150 result = [] 

151 if versioned: 

152 if not self.is_versioned(): 

153 raise Exception('%s is not versioned!' % self.drs_structure) 

154 iter_parts = self.getDrsStructure()['parts_versioned_dataset'] 

155 else: 

156 iter_parts = self.getDrsStructure()['parts_dataset'] 

157 

158 for key in iter_parts: 

159 if key in self.dict['parts']: 

160 result.append(self.dict['parts'][key]) 

161 elif key in self.getDrsStructure()['defaults']: 

162 result.append(self.getDrsStructure()['defaults'][key]) 

163 if to_path: 

164 return self.getDrsStructure()['root_dir'] + '/' + '/'.join(result) 

165 else: 

166 return '.'.join(result)

256 def from_path(path, drs_structure=None): 

257 """Extract a DRSFile object out of a path. 

258:param path: path to a file that is part of the ``drs_structure``. 

259:type param: str 

260:param drs_structure: Which structure is going to be used with this file. 

261:type drs_structure: key value of :class:`DRSFile.DRS_STRUCTURE` 

262""" 

263 if drs_structure is None: 

264 drs_structure = DRSFile.find_structure_from_path(path) 

265 path = os.path.abspath(path) 

266 bl = DRSFile._getDrsStructure(drs_structure) 

267 

268 # trim root_dir 

269 if not path.startswith(bl['root_dir'] + '/'): 

270 raise Exception("File %s does not correspond to %s" % (path, drs_structure)) 

271 

272 parts = path[len(bl['root_dir'])+1:].split('/') 

273 

274 # check the number of parts 

275 if len(parts) != len(bl['parts_dir']): 

276 raise Exception("Can't parse this path. Expected %d elements but got %d." % (len(bl['parts_dir']), len(parts))) 

277 

278 # first the dir 

279 result = {} 

280 result['root_dir'] = bl['root_dir'] 

281 result['parts'] = {} 

282 for i in range(len(bl['parts_dir'])): 

283 result['parts'][bl['parts_dir'][i]] = parts[i] 

284 

285 # split file name 

286 # (extract .nc before splitting) 

287 parts = result['parts']['file_name'][:-3].split('_') 

288 if len(parts) == len(bl['parts_file_name']) - 1 \ 

289 and 'fx' in parts: 

290 # no time 

291 parts.append(None) 

292 

293 try: 

294 for i in range(len(bl['parts_file_name'])): 

295 if bl['parts_file_name'][i] not in result['parts']: 

296 result['parts'][bl['parts_file_name'][i]] = parts[i] 

297 

298 except IndexError: 

299 raise Exception("File %s does not follow the expected naming scheme for %s" % (path, drs_structure)) 

300 

301 bl_file = DRSFile(result, drs_structure=drs_structure) 

302 

303 return bl_file

341 """DEPRECATED: Use solr_search instead! 

342 Simple search for files. It searches locally on the file system using :py:func:`glob.iglob`. 

343This means the values might contain jokers like '\*1960*'. 

344 

345:param drs_structure: name of a DRS structure (key of :class:`DRSFile.DRS_STRUCTURE`) 

346:type drs_structure: str 

347:param latest_version: if this should be only the latest version available. 

348:type latest_version: bool 

349:param partial_dict: a dictionary with some DRS components representing the query. 

350:returns: Generator returning matching files""" 

351 log.warning("Deprecated: User solr_search instead") 

352 bl = DRSFile._getDrsStructure(drs_structure) 

353 search_dict = bl['defaults'].copy() 

354 search_dict.update(partial_dict) 

355 log.debug("Searching in %s using %s", drs_structure, search_dict) 

356 

357 # only baseline 0 is versioned 

358 if latest_version and ('parts_versioned_dataset' not in bl): 

359 latest_version = False 

360 log.debug("No version information stored in the selected structure thus latest version is inactive.") 

361 

362 local_path = bl['root_dir'] 

363 for key in bl['parts_dir']: 

364 if key in search_dict: 

365 local_path = os.path.join(local_path, search_dict[key]) 

366 del search_dict[key] # remove it so we can see if all keys matched 

367 else: 

368 local_path = os.path.join(local_path, "*") 

369 

370 # NOTE: We might have defaults for the datasets that are not appearing in the directories. 

371 if set(search_dict) - set(bl['defaults']): 

372 # ok, there are typos or non existing constraints in the search. 

373 # which are not in the defaults. Those are "strange" to the selected structure. 

374 mesg = "Unknown parameter(s) %s for %s." % (','.join(search_dict), drs_structure) 

375 similar_words = set()

380 mesg = "%s\n\nFor %s try one of: %s" % (mesg, drs_structure, ','.join(bl['parts_dir']))

382 raise Exception(mesg)

389

406 if latest_version: 

407 # then return the results stored in datasets 

408 for latest_version_file in [v for sub in datasets.values() for v in sub]:

411 else:
