35def test_template_dict(): 

36 from evaluation_system.misc.utils import TemplateDict 

37 t = TemplateDict(a=1,b=2,c='$a') 

38 res = t.substitute(dict(x='$b$b$b', y='$z', z='$a'), recursive=False) 

39 assert res == {'x': '222', 'y': '$z', 'z': '1'} 

40 res = t.substitute(dict(x='$b$b$b', y='$c', z='$a'), recursive=True) 

41 assert res == {'x': '222', 'y': '1', 'z': '1'} 

42 tmp = {} 

43 # the maximal amount depends on de order they get resolved, and in  

44 # dictionaries this order is not 

45 # even given by anything in particular (e.g. alphabetic). 

46 # for a transitive substitution (a=b,b=c,c=d,...) the best case is 

47 # always 1 and the worst is ceil(log_2(n)) 

48 # We have a maximal_iterations of 15 so we can substitute *at  

49 # least* 2^15=32768 variables 

50 max_it = 5000 

51 for l in range(max_it): 

52 tmp['a_%s' % l] = '$a_%s' % (l+1) 

53 tmp['a_%s' % max_it] = 'end' 

54 res = t.substitute(tmp, recursive=True) 

55 assert all([r == 'end' for r in res.values()]) 

56 # check recursions that doesn't work 

57 tmp['a_%s' % max_it] = '$a_0' 

58 with pytest.raises(Exception): 

59 t.substitute(tmp, recursive=True) 

60 with pytest.raises(Exception): 

61 t.substitute(dict(x='$y', y='$x'), 

62 recursive=True) 

63 with pytest.raises(Exception): 

64 t.substitute(dict(x='$y', y='$z', z='$x'), 

65 recursive=True)

67def test_metadict_creation(): 

68 from evaluation_system.misc.utils import metadict 

69 m1 = metadict(dict(a=1, b=2, c=[1, 2, 3])) 

70 m2 = metadict(a=1, b=2, c=[1, 2, 3]) 

71 assert m1 == m2 

72 m3 = metadict(a=1, b=2, c=[1, 2, 3]) 

73 m3.setMetadata('a', test=1) 

74 # metadata is just a parallel storage and should not affect the data. 

75 assert m1 == m3 

76 # the 'compact_creation' is a special key! 

77 m4 = metadict(compact_creation=False, a=1, b=2, c=[1, 2, 3]) 

78 assert m1 == m4 

79 assert not ('compact_creation' in m4) 

80 # but after creation you should be able to use it 

81 m4['compact_creation'] = True 

82 assert not (m1 == m4) 

83 assert 'compact_creation' in m4 

84 # setting compact creation to True should only affect tuples! Not lists. 

85 m5 = metadict(compact_creation=True, a=1, b=2, c=[1, 2, 3]) 

86 assert m1 == m5 

87 # Should fail if compact_creation is set and values are bad formed ( 

88 # i.e. iff tuple then (value, dict) 

89 with pytest.raises(AttributeError): 

90 metadict(compact_creation=True, 

91 a=(1, 2), b=2, c=[1, 2, 3]) 

92 with pytest.raises(AttributeError): 

93 metadict(compact_creation=True, 

94 a=(1, [2, 3]), b=2, c=[1, 2, 3]) 

95 # Compact creation should produce the same outcome as the normal one 

96 m6 = metadict(compact_creation=True, a=(1, dict(test=1)), b=2, c=[1, 2, 3]) 

97 assert m1 == m6 

98 assert m3.getMetadata('a') == m6.getMetadata('a')
