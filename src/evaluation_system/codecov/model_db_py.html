77 """Store a an analysis run into the DB. 

78 

79:type tool: :class:`evaluation_system.api.plugin.pluginAbstract` 

80:param tool: the plugin for which we are storing the information. 

81:param config_dict: dictionary with the configuration used for this run, 

82:param uid: the user id (useful in a global database) 

83:param status: the process status 

84:param result: dictionary with the results (created files). 

85""" 

86 if result is None: 

87 result = {} 

88 if slurm_output is None: 

89 slurm_output = 0 

90 if flag is None: 

91 flag = 0 

92 if version_details is None: 

93 version_details = 1 

94 

95 toolname = tool.__class__.__name__.lower() 

96 version = repr(tool.__version__) 

97 

98 newentry = hist.History(timestamp=datetime.now(), 

99 tool=toolname, 

100 version=version, 

101 configuration=json.dumps(config_dict), 

102 slurm_output=slurm_output, 

103 uid_id=uid, 

104 status=status, 

105 flag=flag, 

106 version_details_id=version_details) 

107 # set caption 

108 if caption: 

109 newentry.caption = caption 

110 

111 tool.__parameters__.synchronize(toolname) 

112 

113 try: 

114 newentry.save() 

115 

116 for p in tool.__parameters__._params.values(): 

117 name = p.name 

118 param = Configuration(history_id_id=newentry.id, 

119 parameter_id_id=p.id, 

120 value=json.dumps(config_dict[name]), 

121 is_default=p.is_default) 

122

124 except Exception as e:

185 def getHistory(self, tool_name=None, limit=-1, since=None, until=None, entry_ids=None, uid=None): 

186 """Returns the stored history (run analysis) for the given tool. 

187 

188:type tool_name: str 

189:param tool_name: name of the tool for which the information will be gathered (if None, then everything is returned). 

190:type limit: int 

191:param limit: Amount of rows to be returned (if < 0, return all). 

192:type since: datetime.datetime 

193:param since: Return only items stored after this date 

194:type until: datetime.datetime 

195:param until: Return only items stored before this date 

196:param entry_ids: ([int] or int) id or list thereof to be selected 

197:returns: ([:class:`HistoryEntry`]) list of entries that match the query. 

198""" 

199 filter_dict = {} 

200 

201 if entry_ids is not None: 

202 if isinstance(entry_ids, int): 

203 entry_ids=[entry_ids] 

204 filter_dict['id__in'] = entry_ids 

205 

206 if tool_name is not None: 

207 filter_dict['tool'] = tool_name 

208 

209 if since is not None: 

210 filter_dict['timestamp__gte'] = since 

211 

212 if until is not None: 

213 filter_dict['timestamp__lte'] = until 

214 

215 if uid is not None: 

216 filter_dict['uid_id'] = uid 

217 

218 o = hist.History.objects.filter(**filter_dict).order_by('-id') 

219 

220 if limit > 0: 

221 o = o[:limit] 

222 

223 return o

276 reg_ex = None

278 # regex to get the relative path

286 for file_name in results:
