43 def __init__(self, uid = None, email = None): 

44 '''Creates a user object for the provided id. If no id is given, a user object for 

45the current user, i.e. the one that started the application, is created instead. 

46 

47:type uid: int 

48:param uid: user id in the local system, if not provided the current user is used. 

49:type email: str 

50:param email: user's email address 

51''' 

52 self._dir_type = config.get(config.DIRECTORY_STRUCTURE_TYPE) 

53 

54 if uid is None: 

55 uid = os.getuid() 

56 

57 self._userdata = None 

58 if isinstance(uid, str): 

59 self._userdata = pwd.getpwnam(uid) 

60 else: 

61 self._userdata = pwd.getpwuid(uid) 

62 

63 if self._userdata is None: 

64 raise Exception("Cannot find user %s" % uid) 

65 

66 if email is None: 

67 self._email = '' 

68 else: 

69 self._email = email

71 self._userconfig = Config() 

72 #try to load teh configuration from the very first time. 

73 self._userconfig.read([User.EVAL_SYS_DEFAULT_CONFIG, os.path.join(self._userdata.pw_dir, User.EVAL_SYS_CONFIG)]) 

74 

75 self._db = UserDB(self) 

76 

77 row_id = self._db.getUserId(self.getName()) 

78 

79 if row_id: 

80 try: 

81 self._db.updateUserLogin(row_id, email) 

82 except: 

83 raise

158 def _getUserDir(self, dir_type, tool = None, create=False): 

159 base_dir = dict(base='', config=User.CONFIG_DIR, cache=User.CACHE_DIR, output=User.OUTPUT_DIR, \ 

160 plots=User.PLOTS_DIR, processes=User.PROCESSES_DIR, \ 

161 scheduler_in=config.get(config.SCHEDULER_INPUT_DIR), \ 

162 scheduler_out=config.get(config.SCHEDULER_OUTPUT_DIR)) 

163 

164 if tool is None: 

165 bd = base_dir[dir_type] 

166 # concatenate relative paths only 

167 if bd and bd[0]=='/': 

168 dir_name = bd 

169 else: 

170 #return the directory where the tool configuration files are stored 

171 dir_name = os.path.join(self._getUserBaseDir(), bd) 

172 else: 

173 #It's too confusing if we create case sensitive directories... 

174 tool = tool.lower() 

175 #return the specific directory for the given tool  

176 dir_name = os.path.join(self._getUserBaseDir(), base_dir[dir_type], tool) 

177 

178 #make sure we have a canonical path 

179 dir_name = os.path.abspath(dir_name) 

180 

181 if create and not os.path.isdir(dir_name): 

182 #we are letting this fail in case of problems. 

183 utils.supermakedirs(dir_name, 0o0755) 

184 

185 return dir_name
