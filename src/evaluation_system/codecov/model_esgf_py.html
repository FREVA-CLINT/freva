15class P2P(object): 

16 """ 

17 Handle the connection to a P2P over the Search API 

18 (check the `Search API documentation <http://esgf.org/wiki/ESGF_Search_API>`_.

20 TYPE = Struct.from_dict({'DATASET': 'Dataset', 'FILE': 'File'}) 

21 """Types of results that will be returned."""

23 __MAX_RESULTS = 1000

29 def __init__(self, node='esgf-data.dkrz.de', api='esg-search/search', wget_api='esg-search/wget', defaults=None): 

30 """Creates a connection to the P2P search API. 

31 

32:param node: the p2p search node to connect to. 

33:type node: str 

34:param api: the url path of the service. 

35:type api: str 

36:param defaults: start the connection with the given defaults used for every search. 

37:type defaults: dict 

38 """ 

39 self.node = node 

40 self.api = api 

41 self.wget_api = wget_api 

42 self.defaults = (defaults or {})

44 def __str__(self): 

45 return 'P2P Search API connected to %s' % self.node 

46 def __repr__(self): 

47 return 'Search API: %s - defaults:%s' % (self.__get_url(), self.defaults)

79 def del_defaults(self, *def_keys): 

80 """Remove all the give defaults values if found. For example:: 

81 

82 del_defaults('institute','model') 

83 

84This would remove all values for both *institute* and *model* leaving everything else intact.""" 

85 for k in def_keys:

97 def __get_url(self): 

98 """Just return the url to this service from the values in this object""" 

99 return 'http://%s/%s?format=application%%2Fsolr%%2Bjson' % (self.node, self.api)

101 def __constraints_to_str(self, constraints, **defaults): 

102 """Transform the constraints passed into a url string. Apply defaults on top of it.""" 

103 proc_const = [] 

104 # work on a copy 

105 constraint_dict = self.defaults.copy() 

106 

107 # apply the object wide defaults 

108 if constraints: 

109 constraint_dict.update(constraints) 

110 

111 # if a method defines some defaults apply them on top 

112 # note this works with positive changes, you can't remove anything (yet) 

113 if defaults: 

114 constraint_dict.update(defaults) 

115 

116 # encode constraints into query 

117 if constraint_dict: 

118 for key, value in constraint_dict.items(): 

119 # negations might have been coded as <property>_not_ if not passed in a dictionary 

120 # just transform them to suffix '!' 

121 if key[-5:] == '_not_': 

122 key = key[:-5] + '!' 

123 if isinstance(value, list): 

124 value = ('&%s=' % key).join(map(str, value)) 

125 proc_const.append('%s=%s' % (key, value)) 

126 

127 return '&'.join(proc_const)

129 def get_wget(self, **constraints): 

130 """:returns: (str) a string containing the wget script that can be used for downloading the selected files""" 

131 query = self.__constraints_to_str(constraints, type=P2P.TYPE.FILE) 

132 request = 'http://%s/%s?%s' % (self.node, self.wget_api, query) 

133 return urllib.request.urlopen(request, None, self._time_out).read()

135 def raw_search(self, query): 

136 """A raw search to the Solr index returning everything we get back. 

137 

138:returns: (dict) the raw result as returned from Solr using the search API.""" 

139 request = '%s&%s' % (self.__get_url(), query) 

140 

141 log.debug(request) 

142 response = json.load(urllib.request.urlopen(request, None, self._time_out)) 

143 

144 return response

146 def search(self, type=TYPE.DATASET, **constraints): 

147 """Issue a query to the P2P Search API from the constraint passed and return just the response 

148(no headers). 

149 

150:param type: type of elements to return. 

151:type type: :class:`P2P.TYPE` 

152:param constraints: dictionary with any P2P Search API constraints. 

153:returns: all documents returned from the P2P Search API.""" 

154 query = self.__constraints_to_str(constraints, type=type) 

155 return self.raw_search(query)['response']

157 def get_datasets_names(self, batch_size=__MAX_RESULTS, **constraints): 

158 """:returns: a list of (datasets, version) tuples. (You can't define "fields" while calling this method).""" 

159 if 'fields' in constraints: 

160 del constraints['fields'] 

161 datasets = set([(d['master_id'], int(d['version'])) for d in self.datasets(

165 def get_datasets(self, **constraints): 

166 """ 

167:returns: a list of dataset names. There's no limit imposed in the method itself, So it will 

168 run out of memory if too many datasets are tried to be retrieved. Use the constraint limit=N to get 

169 just the first "N" records. 

170:param constraints: dictionary with any P2P Search API constraints. 

171""" 

172 

173 return [d for d in self.datasets(**constraints)]

175 def datasets(self, batch_size=__MAX_RESULTS, **constraints): 

176 """:returns: a generator iterating through the retrieve Solr docs. 

177:param batch_size: defines the size of the retrieved batch. It means a query will  

178 be sent to Solr every ``batch_size`` entries from the result set. 

179:type batch_size: int 

180:param constraints: dictionary with any P2P Search API constraints.""" 

181 # if we have a limit use it also here. 

182 max_items = sys.maxsize 

183 if 'limit' in constraints: 

184 max_items = int(constraints['limit']) 

185 else: 

186 # if not set the query limit to the batch size 

187 constraints['limit'] = batch_size 

188 if 'offset' in constraints: 

189 sofar = int(constraints['offset']) 

190 del constraints['offset'] 

191 else: 

192 sofar = 0 

193 

194 result = self.search(offset=sofar, **constraints) 

195 

196 total = min(max_items, result['numFound']) 

197 retrieved = len(result['docs']) 

198 for d in result['docs']: 

199 yield d 

200 

201 while sofar + retrieved < total: 

202 result = self.search(offset=sofar+retrieved, **constraints) 

203 total = min(result['numFound'], max_items) 

204 for d in result['docs']: 

205 yield d 

206 sofar += retrieved 

207 retrieved = len(result['docs'])

216 def get_facets(self, *facets, **constraints): 

217 """ 

218:param facets: list of facets to query for. 

219:param constraints: dictionary with any P2P Search API constraints. 

220:returns: ([{facet:{facet_value:count}}]) the facets count from the server.""" 

221 if 'limit' in constraints: 

222 constraints = constraints.copy() 

223 del constraints['limit'] 

224 

225 # facets encoded as a list 

226 if len(facets) == 1 and isinstance(facets[0], list): 

227 facets = facets[0] 

228 query = '%s&limit=0&facets=%s' % (self.__constraints_to_str(constraints), ','.join(facets)) 

229 

230 response = self.raw_search(query)['facet_counts']['facet_fields'] 

231 result = {} 

232 for facet in response: 

233 result[facet] = dict(zip(response[facet][0::2], response[facet][1::2])) 

234 return result
