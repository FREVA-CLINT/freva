1""" 

2.. moduleauthor:: Sebastian Illing / estani 

3 

4This module manages the loading and access to all plug-ins. It is designed as a central registration thorugh 

5which plug-ins can be accessed. 

6 

7Plug-ins are automatically registered if found either in a specific directory ``<evaluation_system_root_dir>/tools`` 

8or by using the environmental variable ``EVALUATION_SYSTEM_PLUGINS``. 

9This variable must to point to the modules implementing :class:`evaluation_system.api.plugin.PluginAbstract`. 

10 

11The value stored there is a colon (':') separated list of ``path,package`` comma (',') separated pairs. The path 

12denotes where the source is to be found, and the package the module name in which the PluginAbstract interface is being 

13implemented. For example:: 

14 

15 EVALUATION_SYSTEM_PLUGINS=/path/to/some/dir/,something.else.myplugin:/other/different/path,some.plugin:/tmp/test,some.other.plugin 

16"""

73def munge(seq): 

74 """Generator to remove duplicates from a list without changing it's order. 

75It's used to keep sys.path tidy. 

76 

77:param seq: any sequence 

78:returns: a generator returning the same objects in the same sequence but skipping duplicates.""" 

79 seen = set() 

80 for item in seq: 

81 if item not in seen: 

82 seen.add(item) 

83 yield item

247def getPluginDict(plugin_name, user_name=User().getName()): 

248 """Return the requested plug-in dictionary entry or raise an exception if not found.

300def parseArguments(plugin_name, arguments, use_user_defaults=False, user=None, config_file=None, check_errors=True): 

301 """Manages the parsing of arguments which are passed as a list of strings. These are in turn 

302sent to an instance of the plugin, that will handle the parsing. This is why the user is required 

303to be known at this stage. 

304 

305:type plugin_name: str 

306:param plugin_name: Name of the plugin to search for. 

307:type arguments: list of strings 

308:param arguments: it will be parsed by the plug-in (see :class:`evaluation_system.api.plugin.parseArguments`) 

309:type use_user_defaults: bool 

310:param use_user_defaults: If ``True`` and a user configuration is found, this will be used as a default for all non 

311set arguments. So the value will be determined according to the first found instance of: argument, 

312user default, tool default 

313:type user: :class:`evaluation_system.model.user.User` 

314:param user: The user for whom this arguments are parsed. 

315:type config_file: str 

316:param config_file: path to a file from where the setup will read a configuration. If None, the default 

317 user dependent one will be used. This will be completely skipped if ``use_user_defaults`` is ``False``. 

318:return: A dictionary with the parsed configuration.""" 

319 plugin_name = plugin_name.lower() 

320 if user is None: 

321 user = User() 

322 

323 p = getPluginInstance(plugin_name, user) 

324 complete_conf = p.__parameters__.parseArguments(arguments, use_defaults=True, check_errors=False) 

325 

326 # if we are using user defaults then load them first 

327 if use_user_defaults: 

328 user_config_file = user.getUserToolConfig(plugin_name) 

329 if os.path.isfile(user_config_file): 

330 with open(user_config_file, 'r') as f: 

331 complete_conf.update(p.readConfiguration(f)) 

332 # now if we still have a config file update what the configuration with it 

333 

334 if isinstance(config_file, str): 

335 if config_file == '-': 

336 # reading from stdin 

337 complete_conf.update(p.readConfiguration(sys.stdin)) 

338 

339 elif config_file is not None: 

340 with open(config_file, 'r') as f: 

341 complete_conf.update(p.readConfiguration(f)) 

342 elif config_file is not None: 

343 # if it's not a string and is something, we assume is something that can be read from 

344 complete_conf.update(p.readConfiguration(config_file)) 

345 

346 # update with user defaults if desired 

347 complete_conf.update(p.__parameters__.parseArguments(arguments, check_errors=False)) 

348 # we haven't check for errors because we might have a half implemented configuration 

349 # some required field might have already been setup (user/system defaults, files, etc) 

350 # but better if we check them 

351 if check_errors: 

352 p.__parameters__.validate_errors(complete_conf, raise_exception=True) 

353 

354 return complete_conf

587 with open(conf_file, 'r') as f: 

588 complete_conf = p.readConfiguration(f) 

589 else: 

590 log.debug('No config file was found in %s', conf_file) 

591 if complete_conf is None:

593 complete_conf = p.setupConfiguration(config_dict=config_dict, recursion=True)

595 log.debug('Running %s with %s', plugin_name, complete_conf)

598

620 result = p._runTool(config_dict=complete_conf, 

621 unique_output=unique_output) 

622 

623 # save results when existing 

624 if result is None: 

625 user.getUserDB().upgradeStatus(rowid, 

626 user.getName(), 

627 History.processStatus.finished_no_output) 

628 

629 else: 

630 # create the preview 

631 preview_path = config.get(config.PREVIEW_PATH, None) 

632

634 logging.debug('Converting....') 

635 _preview_create(plugin_name, result) 

636 logging.debug('finished') 

637 

638 # write the created files to the database 

639 logging.debug('Storing results into data base....') 

640 user.getUserDB().storeResults(rowid, result) 

641 logging.debug('finished') 

642 

643 # temporary set all processes to finished 

644 user.getUserDB().upgradeStatus(rowid, 

645 user.getName(), 

646 History.processStatus.finished)

657def scheduleTool(plugin_name, slurmoutdir=None, config_dict=None, user=None, 

658 caption=None, unique_output=True): 

659 """Schedules a tool and stores this "run" in the :class:`evaluation_system.model.db.UserDB`. 

660 

661:type plugin_name: str 

662:param plugin_name: name of the referred plugin. 

663:type slurmoutdir: string 

664:param slurmoutdir: directory for the output 

665:type config_dict: dict or metadict 

666:param config_dict: The configuration used for running the tool. If is None, the default configuration will be stored, 

667 this might be incomplete. 

668:type user: :class:`evaluation_system.model.user.User` 

669:param user: The user starting the tool 

670:type scheduled_id: int 

671:param scheduled_id: if the process is already scheduled then put the row id here 

672:type caption: str 

673:param caption: the caption to set. 

674""" 

675 

676 plugin_name = plugin_name.lower() 

677 if user is None: 

678 user = User() 

679 

680 p = getPluginInstance(plugin_name, user) 

681 complete_conf = None 

682 

683 # check whether a scheduled id is given 

684 if config_dict is None: 

685 conf_file = user.getUserToolConfig(plugin_name) 

686 if os.path.isfile(conf_file): 

687 log.debug('Loading config file %s', conf_file) 

688 with open(conf_file, 'r') as f: 

689 complete_conf = p.readConfiguration(f) 

690 else: 

691 log.debug('No config file was found in %s', conf_file) 

692 if complete_conf is None: 

693 # at this stage we want to resolve or tokens and perform some kind of sanity check before going further 

694 complete_conf = p.setupConfiguration(config_dict=config_dict, recursion=True) 

695 

696 log.debug('Schedule %s with %s', plugin_name, complete_conf) 

697 

698 slurmindir = os.path.join(user.getUserSchedulerInputDir(), user.getName()) 

699 if not os.path.exists(slurmindir): 

700 utils.supermakedirs(slurmindir, 0o0777) 

701 

702 version_details = getVersion(plugin_name) 

703 rowid = user.getUserDB().storeHistory(p, 

704 complete_conf, 

705 user.getName(), 

706 History.processStatus.not_scheduled, 

707 version_details=version_details, 

708 caption=caption) 

709 

710 # follow the notes 

711 followHistoryTag(rowid, user.getName(), 'Owner') 

712 

713 # set the SLURM output directory 

714 if not slurmoutdir: 

715 slurmoutdir = user.getUserSchedulerOutputDir() 

716 slurmoutdir = os.path.join(slurmoutdir, plugin_name) 

717 

718 if not os.path.exists(slurmoutdir): 

719 utils.supermakedirs(slurmoutdir, 0o0777) 

720 

721 # write the SLURM file 

722 p.rowid = rowid 

723 full_path = os.path.join(slurmindir, p.suggestSlurmFileName()) 

724 with open(full_path, 'w') as fp: 

725 p.writeSlurmFile(fp, 

726 scheduled_id=rowid, 

727 user=user, 

728 slurmoutdir=slurmoutdir, 

729 unique_output=unique_output) 

730 

731 # create the batch command 

732 command = ['/bin/bash', 

733 '-c', 

734 '%s %s %s\n' % (config.get('scheduler_command'), # SCHEDULER_COMMAND, 

735 config.SCHEDULER_OPTIONS, 

736 #user.getName(), 

737 full_path)] 

738 

739 # run this 

740 logging.debug("Command: " + str(command)) 

741 res = run(command, stdout=PIPE, stderr=PIPE) 

742 stdout, stderr = res.stdout.decode(), res.stderr.decode() 

743 logging.debug("scheduler call output:\n" + str(stdout)) 

744 logging.debug("scheduler call error:\n" + str(stderr)) 

745 

746 # get the very first line only 

747 out_first_line = stdout.split('\n')[0] 

748 

749 # read the id from stdout 

750 if out_first_line.split(' ')[0] == 'Submitted': 

751 slurm_id = int(out_first_line.split(' ')[-1]) 

752 else: 

753 slurm_id = 0 

754 raise Exception('Unexpected scheduler output:\n%s' % out_first_line) 

755 

756 slurm_out = os.path.join(slurmoutdir, 

757 'slurm-%i.out' % slurm_id) 

758 

759 # create a standard slurm file to view with less 

760 with open(slurm_out, 'w') as the_file: 

761 the_file.write('Certainly, your job is pending with id %i.\n' % slurm_id) 

762 the_file.write('You can get further information using the command squeue.\n') 

763 the_file.write('\nThis file was automatically created by the evaluation system.\n') 

764 the_file.write('It will be overwritten by the output of %s.\n' % plugin_name) 

765 

766 # set the slurm output file 

767 user.getUserDB().scheduleEntry(rowid, user.getName(), slurm_out) 

768 

769 return rowid, slurm_out

810def getCommandStringFromRow(history_row, command_name='analyze', command_options='--tool'): 

811 """ 

812 Return the parameter string of a history entry. 

813 :type history_row: row 

814 :param history_row: row of the history table 

815 """ 

816 

817 result = "%s %s %s" % (command_name, command_options, history_row.tool) 

818 

819 configuration = history_row.config_dict() 

820 

821 # find lists 

822 re_list_pattern = r"^\[.*\]$" 

823 re_list = re.compile(re_list_pattern) 

824 

825 for k in configuration.keys(): 

826 value = configuration[k] 

827 

828 if value is not None: 

829 # convert non-None values to string 

830 value = str(value) 

831 

832 # remove brackets from list 

833 if re_list.match(value):

835 

836 result = "%s %s='%s'" % (result, str(k), value) 

837
