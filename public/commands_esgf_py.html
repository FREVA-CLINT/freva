40 __short_description__ = '''Browse ESGF data and create wget script''' 

41 __description__ = """ 

42The query is of the form key=value. the key might be repeated and/or negated with the  

43'_not_' suffix (e.g. model_not_=MPI-ESM-LR experiment=decadal2000 experiment=decadal2001) 

44 

45Simple query: 

46 freva --esgf model=MPI-ESM-LR experiment=decadal2001 variable=tas distrib=False 

47  

48The search API is described here: http://www.esgf.org/wiki/ESGF_Search_REST_API 

49Some special query keys: 

50distrib: (*true*, false) search globally or only at DKRZ (MPI data and replicas) 

51latest : (true, false, *unset*) search for the latest version, older ones or all. 

52replica: (true, false, *unset*) search only for replicas, non-replicas, or all. 

53""" 

54 

55 def _run(self): 

56 """Command line options.""" 

57 args = self.args 

58 lastargs = self.last_args 

59 DEBUG = self.DEBUG 

60 

61 url_type = 'http' 

62 if args.opendap: 

63 url_type = 'opendap' 

64 if args.gridftp: 

65 url_type = 'gridftp' 

66 

67 # defaults 

68 datasets = args.datasets 

69 facets = {} 

70 show_facets = [] 

71 if args.show_facet: 

72 show_facets.append(args.show_facet) 

73 download_file = args.download_script 

74 query = args.query 

75 

76 for arg in lastargs: 

77 items = arg.split('=') 

78 if items[0] not in facets: 

79 facets[items[0]] = [] 

80 if len(items) > 1: 

81 facets[items[0]].append('='.join(items[1:])) 

82 

83 if DEBUG: 

84 sys.stderr.write("Searching string: %s\n" % facets) 

85 

86 # flush stderr in case we have something pending 

87 sys.stderr.flush() 

88 

89 # find the files and display them 

90 p2p = P2P() 

91 

92 if download_file: 

93 download_file = Path(download_file) 

94 download_file.touch(0o755) 

95 with download_file.open('bw') as f: 

96 f.write(p2p.get_wget(**facets)) 

97 print(f"Download script successfully saved to {download_file}") 

98 return 0 # there's nothing more to be executed after this 

99 return 1 

100 if datasets: 

101 print('\n'.join(['%s - version: %s' % d for d in sorted(p2p.get_datasets_names(**facets))])) 

102 if query: 

103 if len(query.split(',')) > 1: 

104 # we get multiple fields queried, return in a tructured fashion 

105 p2p.show(p2p.get_datasets(fields=query,**facets)) 

106 else: 

107 # if only one then return directly 

108 print('\n'.join([str(d[query]) for d in p2p.get_datasets(fields=query, **facets)])) 

109 

110 if show_facets: 

111 results = p2p.get_facets(show_facets, **facets) 

112 # render them 

113 for facet_key in sorted(results): 

114 if len(results[facet_key]) == 0: 

115 values = "<No Results>" 

116 else: 

117 values = '\n\t'.join(['%s: %s' % (k, results[facet_key][k]) for k in sorted(results[facet_key])]) 

118 print('[%s]\n\t%s' % (facet_key, values)) 

119 if not (datasets or query or show_facets): 

120 # default 

121 for result in p2p.files(fields='url', **facets): 

122 for url_encoded in result['url']: 

123 url, _, access_type = url_encoded.split('|') 

124 if access_type.lower().startswith(url_type): 

125 print(url) 

126 

127if __name__ == "__main__": # pragma nocover 

128 Command().run()
