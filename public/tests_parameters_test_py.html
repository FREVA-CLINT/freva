121def test_parse_arguments(dummy_env): 

122 from evaluation_system.api.parameters import (String, ParameterDictionary, 

123 Integer, Float, Bool, String, File, 

124 Date, Range, ValidationError) 

125 p_dict = ParameterDictionary(String(name='a'), String(name='b')) 

126 res = p_dict.parseArguments("a=1 b=2".split()) 

127 assert res == dict(a='1', b='2') 

128 p_dict = ParameterDictionary(Integer(name='a'),Integer(name='b')) 

129 res = p_dict.parseArguments("a=1 b=2".split()) 

130 assert res == dict(a=1, b=2) 

131 # more arguments than those expected 

132 p_dict = ParameterDictionary(Integer(name='a')) 

133 with pytest.raises(ValidationError): 

134 p_dict.parseArguments("a=1 b=2".split()) 

135 p_dict = ParameterDictionary(Integer(name='int'), 

136 Float(name='float'), 

137 Bool(name='bool'), 

138 String(name='string'), 

139 File(name='file', default='/tmp/file1'), 

140 Date(name='date'), 

141 Range(name='range')) 

142 res = p_dict.parseArguments("int=1 date=1 bool=1".split()) 

143 assert res == dict(int=1, date='1', bool=True) 

144 res = p_dict.parseArguments("int=1 date=1 bool=1".split(), 

145 use_defaults=True) 

146 assert res == dict(int=1, date='1', bool=True, file='/tmp/file1') 

147 

148 res = p_dict.parseArguments( 

149 "int=1 date=1 bool=1 range=1:5".split(), 

150 use_defaults=True, complete_defaults=True 

151 ) 

152 assert res == dict(int=1, date='1', bool=True, 

153 range=list(range(1,6)), file='/tmp/file1', 

154 float=None, string=None) 

155 

156 for arg, parsed_val in [("bool=1",True), ("bool=true",True), 

157 ("bool=TRUE",True), 

158 ("bool=0",False), ("bool=false",False), 

159 ("bool=False",False), 

160 ("bool=no",False), ("bool=NO",False), 

161 ("bool=YES",True), 

162 ("bool", True), #Special case! 

163 ("float=1.2", 1.2), ("float=-1e-1",-0.1), 

164 ("float=2E2",200.0), 

165 ("string=1", "1"), ("string=ä", "ä")]: 

166 res = p_dict.parseArguments(arg.split()) 

167 assert res == {arg.split('=')[0]:parsed_val}, \ 

168 'Error when parsing %s, got %s' % (arg, res) 

169 

170 # multiple arguments 

171 p_dict = ParameterDictionary(File(name='file', default='/tmp/file1', 

172 max_items=2, item_separator=':'), 

173 Date(name='date', item_separator='/')) 

174 assert p_dict.parseArguments(["file=a:b"]) == \ 

175 dict(file=['a','b']) 

176 with pytest.raises(ValidationError): 

177 p_dict.parseArguments(["file=a:b:c"]) 

178 with pytest.raises(ValidationError): 

179 p_dict.parseArguments(["file=a","file=b", "file=c"]) 

180 # this should still work since max_items defaults to 1  

181 # and in that case no splitting happens 

182 assert p_dict.parseArguments(["date=a/b"]) == \ 

183 dict(date="a/b") 

184 assert p_dict.parseArguments(["file=a", "file=b"]) == \ 

185 dict(file=['a','b'])
