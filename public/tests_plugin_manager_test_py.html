172plugin = modules['evaluation_system.api.plugin']

175class %s(plugin.PluginAbstract): 

176 __short_description__ = "Test" 

177 __version__ = (0,0,1) 

178 __parameters__ = parameters.ParameterDictionary( 

179 parameters.File(name="output", default="/tmp/file", help='output'), 

180 parameters.File(name="input", mandatory=True, help="some input"))

182 def runTool(self, config_dict=None): 

183 print("%s", config_dict)"""
