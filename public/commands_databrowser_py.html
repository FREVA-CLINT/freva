17class Command(FrevaBaseCommand): 

18 

19 _args = [ 

20 {'name': '--debug', 'short': '-d', 'help': 'turn on debugging info and show stack trace on exceptions.', 

21 'action': 'store_true'}, 

22 {'name': '--help', 'short': '-h', 'help': 'show this help message and exit', 'action': 'store_true'}, 

23 {'name': '--multiversion', 'help': 'select not only the latest version but all of them', 

24 'action': 'store_true', 'default': False}, 

25 {'name': '--relevant-only', 'help': 'show only facets that filter results (i.e. >1 possible values)', 

26 'action': 'store_true'}, 

27 {'name': '--batch-size', 'help': 'Number of files to retrieve', 'type': 'int', 'metavar': 'N'}, 

28 {'name': '--count-facet-values', 'help': 'Show the number of files for each values in each facet', 

29 'action': 'store_true'}, 

30 {'name': '--attributes', 'help': 'retrieve all possible attributes for the current search instead of the files', 

31 'action': 'store_true'}, 

32 {'name': '--all-facets', 'help': 'retrieve all facets (attributes & values) instead of the files', 

33 'action': 'store_true'}, 

34 {'name': '--facet', 'help': 'retrieve these facets (attributes & values) instead of the files'}, 

35 ] 

36 

37 __description__ = """ 

38The query is of the form key=value. <value> might use *, ? as wildcards or any regular expression enclosed in forward slashes. Depending on your shell and the symbols used, remember to escape the sequences properly. 

39The safest would be to enclosed those in single quotes. 

40 

41For Example: 

42 %s project=baseline1 model=MPI-ESM-LR experiment=/decadal200[0-3]/ time_frequency=*hr variable='/ta|tas|vu/'""" 

43 

44 __short_description__ = '''Find data in the system''' 

45 

46 def _run(self): 

47 

48 args = self.args 

49 last_args = self.last_args 

50 

51 # Are we searching for facets or files? 

52 facets = [] 

53 if args.all_facets: 

54 facets = None 

55 if args.facet: 

56 facets.append(args.facet) 

57 

58 latest = not args.multiversion 

59 batch_size = args.batch_size if args.batch_size else 10 

60 

61 search_dict = {} 

62 # contruct search_dict by looping over last_args 

63 for arg in last_args: 

64 if '=' not in arg: 

65 raise CommandError("Invalid format for query: %s" % arg) 

66 

67 items = arg.split('=') 

68 key, value = items[0], ''.join(items[1:]) 

69 

70 if key not in search_dict: 

71 search_dict[key] = value 

72 else: 

73 if not isinstance(search_dict[key], list): 

74 search_dict[key] = [search_dict[key]] 

75 search_dict[key].append(value) 

76 

77 if 'version' in search_dict and latest: 

78 # it makes no sense to look for a specific version just among the latest 

79 # the speedup is marginal and it might not be what the user expects 

80 sys.stderr.write('Turning latest of when searching for a specific version.') 

81 latest = False 

82 

83 logging.debug("Searching dictionary: %s\n", search_dict) 

84 # exit() 

85 # flush stderr in case we have something pending 

86 sys.stderr.flush() 

87 

88 if facets != [] and not args.attributes: 

89 if 'facet.limit' in search_dict: 

90 facet_limit = int(search_dict['facet.limit']) 

91 else: 

92 # default 

93 facet_limit = 1000 

94 search_dict['facet.limit'] = -1 

95 

96 for att, values in SolrFindFiles.facets(facets=facets, latest_version=latest, **search_dict).items(): 

97 # values come in pairs: (value, count) 

98 value_count = len(values) // 2 

99 if args.relevant_only and value_count < 2: 

100 continue 

101 

102 if args.count_facet_values: 

103 sys.stdout.write('%s: %s' % (att, ','.join(['%s (%s)' % (v, c) for v, c in zip(*[iter(values)]*2)]))) 

104 else: 

105 sys.stdout.write('%s: %s' % (att, ','.join(values[::2]))) 

106 

107 if value_count == facet_limit: 

108 sys.stdout.write('...') 

109 

110 sys.stdout.write('\n') 

111 sys.stdout.flush() 

112 elif args.attributes: 

113 # select all is none defined but this flag was set 

114 if not facets: 

115 facets = None 

116 results = SolrFindFiles.facets(facets=facets, latest_version=latest, **search_dict) 

117 if args.relevant_only: 

118 atts = ', '.join([k for k in results if len(results[k]) > 2]) 

119 else: 

120 atts = ', '.join(SolrFindFiles.facets(facets=facets, latest_version=latest, **search_dict)) 

121 sys.stdout.write(atts) 

122 sys.stdout.write('\n') 

123 sys.stdout.flush() 

124 else: 

125 # find the files and display them 

126 for f in SolrFindFiles.search(batch_size=batch_size, latest_version=latest, **search_dict): 

127 sys.stdout.write(str(f)) 

128 sys.stdout.write('\n') 

129 sys.stdout.flush() 

130 

131if __name__ == "__main__": # pragma nocover 

132 Command().run()
