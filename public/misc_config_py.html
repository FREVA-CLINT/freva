131def reloadConfiguration(): 

132 """Reloads the configuration. 

133This can be used for reloading a new configuration from disk.  

134At the present time it has no use other than setting different configurations  

135for testing, since the framework is restarted every time an analysis is  

136performed.""" 

137 global _config 

138 _config = { BASE_DIR:'evaluation_system', 

139 BASE_DIR_LOCATION: os.path.expanduser('~'), 

140 DIRECTORY_STRUCTURE_TYPE: DIRECTORY_STRUCTURE.LOCAL, 

141 PLUGINS: {}} 

142 

143 #now check if we have a configuration file, and read the defaults from there 

144 config_file = os.environ.get(_DEFAULT_ENV_CONFIG_FILE, 

145 _DEFAULT_CONFIG_FILE_LOCATION) 

146 

147 #print os.environ.get('EVALUATION_SYSTEM_CONFIG_FILE', None), 'the config file' 

148 log.debug("Loading configuration file from: %s", config_file) 

149 if config_file and os.path.isfile(config_file): 

150 config_parser = ConfigParser() 

151 with open(config_file, 'r') as fp: 

152 config_parser.read_file(fp) 

153 if not config_parser.has_section(CONFIG_SECTION_NAME): 

154 raise ConfigurationException(("Configuration file is missing section %s.\n" 

155 + "For Example:\n[%s]\nprop=value\n...") \ 

156 % (CONFIG_SECTION_NAME, CONFIG_SECTION_NAME)) 

157 else: 

158 _config.update(config_parser.items(CONFIG_SECTION_NAME)) 

159 

160 for plugin_section in [s for s in config_parser.sections() if s.startswith(PLUGINS)]: 

161 _config[PLUGINS][plugin_section[len(PLUGINS):]] = \ 

162 SPECIAL_VARIABLES.substitute(dict(config_parser.items(plugin_section))) 

163 

164 

165 log.debug('Configuration loaded from %s', config_file) 

166 else: 

167 log.debug('No configuration file found in %s. Using default values.', 

168 config_file) 

169 

170 _config = SPECIAL_VARIABLES.substitute(_config, recursive=False) 

171 

172 #perform all special checks 

173 if not DIRECTORY_STRUCTURE.validate(_config[DIRECTORY_STRUCTURE_TYPE]): 

174 raise ConfigurationException("value (%s) of %s is not valid. Should be one of: %s" \ 

175 % (_config[DIRECTORY_STRUCTURE_TYPE], DIRECTORY_STRUCTURE_TYPE, 

176 ', '.join(DIRECTORY_STRUCTURE.toDict().values()))) 

177#load the configuration for the first time 

178reloadConfiguration()

180_nothing = object() 

181def get(config_prop, default=_nothing): 

182 """Returns the value stored for the given config_prop. 

183If the config_prop is not found and no default value is provided an exception 

184will be thrown. If not the default value is returned. 

185 

186:param config_prop: property for which it's value is looked for. 

187:type config_prop: str 

188:param default: If the property is not found this value is returned. 

189:return: the value associated with the given property, the default one  

190if not found or an exception is thrown if no default is provided. 

191""" 

192 

193 if config_prop in _config: 

194 return _config[config_prop] 

195 elif default != _nothing: 

196 return default 

197 else: 

198 raise ConfigurationException("No configuration for %s" % config_prop) 

199 

200def get_plugin(plugin_name, config_prop, default=_nothing): 

201 """Returns the value stored for the given config_prop at the given plugin. 

202If the config_prop is not found and no default value is provided an exception 

203will be thrown. If not the default value is returned. If the plug.in name does 

204not exists an exception is thrown. 

205 

206:param plugin_name: name of an existing plug-in as returned by  

207:class:`get`(:class:`PLUGINS`) 

208:type plugin_name: str 

209:param config_prop: property for which it's value is looked for. 

210:type config_prop: str 

211:param default: If the property is not found this value is returned. 

212:return: the value associated with the given property, the default one  

213if not found  

214 or an exception is thrown if no default is provided. 

215""" 

216 _plugin = _config[PLUGINS] 

217 if plugin_name in _config[PLUGINS]: 

218 _plugin_config = _config[PLUGINS][plugin_name] 

219 if config_prop in _plugin_config: 

220 return _plugin_config[config_prop] 

221 elif default != _nothing: 

222 return default 

223 else: 

224 raise ConfigurationException("No configuration for %s for plug-in %s" % (config_prop, plugin_name)) 

225 else: 

226 raise ConfigurationException("No plug-in named %s" % plugin_name)
