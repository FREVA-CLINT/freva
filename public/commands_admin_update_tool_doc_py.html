20 

21class Command(FrevaBaseCommand): 

22 

23 _args = [ 

24 {'name': '--debug', 'short': '-d', 

25 'help': 'turn on debugging info and show stack trace on exceptions.', 'action': 'store_true'}, 

26 {'name': '--help', 'short': '-h', 

27 'help': 'show this help message and exit', 'action': 'store_true'}, 

28 {'name': '--docpath', 'help': 'path to doc folder with tex file'}, 

29 {'name': '--tex_file', 'help': 'filename of main tex file'}, 

30 {'name': '--tool', 'help': 'tool name'} 

31 ]

33 __short_description__ = '''Update the html files of tool documentation'''

35 def copy_and_overwrite(self, from_path, to_path): 

36 if os.path.exists(to_path): 

37 shutil.rmtree(to_path) 

38 if os.path.exists(from_path): 

39 shutil.copytree(from_path, to_path) 

40 

41 def _run(self): 

42 

43 doc_path = self.args.docpath 

44 tex_file = self.args.tex_file 

45 tool = self.args.tool.lower() 

46 if not tex_file: 

47 print('Can\'t find a .tex file in this directory!') 

48 return 

49 file_root = tex_file.split('.')[0] 

50 # copy folder to /tmp for processing 

51 new_path = '/tmp/%s/' % tool 

52 self.copy_and_overwrite(doc_path, new_path) 

53 

54 # change path and run "htlatex" and "bibtex" 

55 os.chdir(new_path) 

56 cfg_file = os.path.dirname(__file__)+'/../../../../etc/ht5mjlatex.cfg' 

57 os.system('htlatex %s "%s"' % (new_path+tex_file, cfg_file)) 

58 os.system('bibtex %s' % file_root) 

59 os.system('htlatex %s "%s"' % (new_path+tex_file, cfg_file)) 

60 

61 # open html file and remove <head> and <body> tags 

62 fi = open(os.path.join(new_path, file_root+'.html')) 

63 text = fi.read() 

64 text = re.sub("<head>.*?</head>", "", text, flags=re.DOTALL) 

65 text = text.replace('</html>', '') 

66 text = text.replace('<html>', '') 

67 text = text.replace('</body>', '') 

68 text = text.replace('<body>', '') 

69 

70 figure_prefix = 'figures' 

71 # replace img src 

72 text = text.replace('src="%s/' % figure_prefix, 

73 'style="width:80%;" src="/static/preview/doc/'+tool+'/') 

74 

75 # remove too big sigma symbols 

76 text = text.replace('mathsize="big"', '') 

77 

78 flat_page, created = FlatPage.objects.get_or_create( 

79 title=self.args.tool, url='/about/%s/' % tool 

80 ) 

81 if created: 

82 flat_page.sites = [1] 

83 flat_page.content = text 

84 flat_page.save() 

85 

86 # Copy images to website preview path 

87 preview_path = config.get('preview_path') 

88 dest_dir = os.path.join(preview_path, 'doc/%s/' % tool) 

89 self.copy_and_overwrite('%s/' % figure_prefix, dest_dir) 

90 if not os.path.exists(dest_dir): 

91 os.makedirs(dest_dir) 

92 shutil.copyfile('%s.css' % tool, os.path.join(dest_dir, '%s.css' % tool)) 

93 # remove tmp files 

94 shutil.rmtree(new_path) 

95 

96 

97if __name__ == "__main__": 

98 Command().run()
